+program textcmd_RefreshVitals(who, text)
+	var targ;
+	if ( text == "me" )
+		targ := who;
+	else
+		SendSysMessage(who, "Select a mobile.");
+		targ := Target(who);
+		if ( !targ.IsA(POLCLASS_MOBILE) )
+			SendSysMessage(who, "Cancelled.");
+			return 0;
+		endif
+	endif
+	
+	AP_RefreshVitals(targ);
+	
+	SendSysMessage(who, "Done.");
+endprogram
